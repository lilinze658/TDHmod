	on_war = {
		effect = {
			if = {
				limit = {
					THIS = {
						NOT = { has_country_flag = AREA_gbpaint_country_flag }

	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = SHE

					}

						OR = {
							original_tag = SHE
							original_tag = ONG
							original_tag = GOD
							original_tag = ANY
							original_tag = SOU
						}
					}
				}
				ROOT = {
					add_timed_idea = {
						idea = SHE_war_plan_black_timed
						days = 90
					}
					clr_country_flag = SHE_war_plan_black_flag
				}
			}
		}
